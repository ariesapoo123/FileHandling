import random

f = open('test-2.txt', 'w')

lines = []

for j in range(11):
    lines.append(random.randint(1, 20))

f.writelines(str(lines))

f.close()


import random
f = open('test-3.txt', 'w')

lines = []

for j in range(11):
    lines.append(random.randint(1, 20))
    lines.sort()
f.writelines(str(lines))

f.close()
